let currentBotId = 'general'; const messagesContainer = document.getElementById('messages-container'); const userInput = document.getElementById('user-input'); const sendBtn = document.getElementById('send-btn'); const typingIndicator = document.getElementById('typing-indicator'); function init() { renderSidebar(); adjustTextareaHeight(); } function renderSidebar() { const list = document.querySelector('.bots-list'); if (!list) return; list.innerHTML = ''; bots.forEach(bot => { const item = document.createElement('div'); item.className = `bot-item ${bot.id === currentBotId ? 'active' : ''}`; item.onclick = () => switchBot(bot.id); item.innerHTML = `

${bot.name}

${bot.desc}

`; list.appendChild(item); }); } function switchBot(id) { currentBotId = id; const bot = bots.find(b => b.id === id); document.getElementById('header-name').textContent = bot.name; document.querySelector('#header-avatar i').className = `fa-solid ${bot.icon}`; renderSidebar(); document.getElementById('sidebar').classList.remove('open'); clearChat(false); addMessage('bot', `تم تفعيل وضع ${bot.name}. ${bot.role}`); } function addMessage(sender, text) { const msgDiv = document.createElement('div'); msgDiv.className = `message ${sender}`; const avatarIcon = sender === 'user' ? 'fa-user' : bots.find(b => b.id === currentBotId).icon; msgDiv.innerHTML = `
${formatText(text)}
`; messagesContainer.appendChild(msgDiv); scrollToBottom(); } function formatText(text) { text = text.replace(/```([\s\S]*?)```/g, '
$1
'); text = text.replace(/\*\*(.*?)\*\*/g, '$1'); return text.replace(/\n/g, '
'); } function scrollToBottom() { messagesContainer.scrollTop = messagesContainer.scrollHeight; } function clearChat(addWelcome = true) { messagesContainer.innerHTML = ''; if (addWelcome) addMessage('bot', 'بدأت محادثة جديدة. كيف يمكنني مساعدتك؟'); } async function sendMessage() { const text = userInput.value.trim(); if (!text) return; addMessage('user', text); userInput.value = ''; userInput.style.height = 'auto'; sendBtn.disabled = true; typingIndicator.style.display = 'block'; scrollToBottom(); try { const response = await fetch('https://api.openai.com/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer YOUR_OPENAI_API_KEY_HERE' // [حط هنا المفتاح ديالك] }, body: JSON.stringify({ model: "gpt-3.5-turbo", messages: [{ role: "user", content: text }] }) }); const data = await response.json(); const aiResponse = data.choices[0].message.content; typingIndicator.style.display = 'none'; addMessage('bot', aiResponse); } catch (error) { typingIndicator.style.display = 'none'; addMessage('bot', 'حدث خطأ. تأكد من مفتاح الـ API.'); } finally { sendBtn.disabled = false; } } userInput.addEventListener('input', adjustTextareaHeight); userInput.addEventListener('keydown', (e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); } }); sendBtn.addEventListener('click', sendMessage); function adjustTextareaHeight() { userInput.style.height = 'auto'; userInput.style.height = userInput.scrollHeight + 'px'; } init();
