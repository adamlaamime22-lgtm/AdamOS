${bot.name}

`; list.appendChild(item); }); } // --- تبديل البوت --- function switchBot(id) { currentBotId = id; const bot = bots.find(b => b.id === id); document.getElementById('header-name').textContent = bot.name; document.querySelector('#header-avatar i').className = `fa-solid ${bot.icon}`; document.querySelector('#header-avatar').style.background = `linear-gradient(135deg, ${bot.color}, #fff)`; document.querySelector('#header-avatar').style.color = '#000'; renderSidebar(); document.getElementById('sidebar').classList.remove('open'); clearChat(false); addMessage('bot', `تم تفعيل وضع ${bot.name}. ${bot.role}`); } // --- إضافة رسالة للشات --- function addMessage(sender, text) { const msgDiv = document.createElement('div'); msgDiv.className = `message ${sender}`; const avatarIcon = sender === 'user' ? ${formatText(text)}
`; messagesContainer.appendChild(msgDiv); scrollToBottom(); } // --- تنسيق النص --- function formatText(text) { if (text.includes('```')) { text = text.replace(/```([\s\S]*?)```/g, '

'); } text = text.replace(/\*\*(.*?)\*\*/g, '$1'); text = text.replace(/\n/g, '
'); return text; } function scrollToBottom() { messagesContainer.scrollTop = messagesContainer.scrollHeight; } function clearChat(addWelcome = true) { messagesContainer.innerHTML = ''; if (addWelcome) { addMessage('bot', 'بدأت محادثة جديدة. كيف يمكنني مساعدتك؟'); } } // --- الربط مع الـ AI الحقيقي (هنا فين كاين "المخ" الحقيقي) --- async function sendMessage() { const text = userInput.value.trim(); if (!text) return; addMessage('user', text); userInput.value = ''; userInput.style.height = 'auto'; sendBtn.disabled = true; typingIndicator.style.display = 'block'; scrollToBottom(); try { const response = await fetch('[https://api.openai.com/v1/chat/completions](https://api.openai.com/v1/chat/completions)', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer YOUR_OPENAI_API_KEY_HERE' // [حط هنا المفتاح ديالك] }, body: JSON.stringify({ model: "gpt-3.5-turbo", messages: [{ role: "user", content: text }] }) }); const data = await response.json(); const aiResponse = data.choices[0].message.content; typingIndicator.style.display = 'none'; addMessage('bot', aiResponse); } catch (error) { typingIndicator.style.display = 'none'; addMessage('bot', 'عذراً، حدث خطأ. تأكد من اتصالك ومن مفتاح الـ API.'); } finally { sendBtn.disabled = false; } } // --- مستمعي الأحداث --- userInput.addEventListener('input', function() { adjustTextareaHeight(); sendBtn.disabled = this.value.trim() === ''; }); userInput.addEventListener('keydown', function(e) { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); } }); sendBtn.addEventListener('click', sendMessage); function adjustTextareaHeight() { userInput.style.height = 'auto'; userInput.style.height = userInput.scrollHeight + 'px'; } function toggleSidebar() { document.getElementById('sidebar').classList.toggle('open'); } init();
